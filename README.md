#Ejercicio 3:
dia = input("Ingrese un dia de la semana: ")
if dia.lower() == "lunes":
    print("Hoy es lunes")
elif dia.lower() == "viernes":
    print("Hoy es viernes")
elif dia.lower() == "sabado" or dia.lower() == "domingo":
    print("Hoy es fin de semana")
else:
    print("Hoy no es un dia especial")
